#include <xc.h>
#define _XTAL_FREQ 20000000
void UART_Init() 
 {
    TRISC6 = 1;
    TRISC7 = 1;
    SPBRG = 129;
    BRGH = 1;
    SYNC = 0;
    SPEN = 1;
    TXEN = 1;
    CREN = 1;
}
void UART_TxChar(char data) {
    while (!TRMT);
    TXREG = data;
}
void UART_TxString(const char *str) {
    while (*str) {
        UART_TxChar(*str++);
    }
}
char UART_RxChar() {
    while (!RCIF);
    return RCREG;
}
